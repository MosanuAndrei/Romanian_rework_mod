
# Fix Moldavia leaving PU Bug
fix_pu_moldavia = {

	is_shown = {
		this = c:WAL
	}

	possible = {
		exists = c:MOL
		c:MOL = {
			is_subject = no
			is_player = no
		}
	}
	
	when_taken = {
		create_diplomatic_pact = {
            country = c:MOL
			target = c:WAL
            type = personal_union
		}
	}
	ai_chance = {
		base = 100
	}
}
